⚠️ Supabase 테이블 설정 필요

Supabase 대시보드에서 SQL Editor를 열고 아래 SQL을 실행해주세요:

create table share_links (
  id bigserial primary key,
  key text unique not null,
  title text,
  expiry date,
  prog_ids jsonb,
  prog_data jsonb,
  created_at timestamptz default now()
);

create table share_requests (
  id bigserial primary key,
  share_key text,
  company text,
  name text,
  phone text,
  description text,
  slots jsonb,
  slot_summary text,
  status text default 'pending',
  created_at timestamptz default now()
);

create table user_data (
  id text primary key,
  payload text,
  updated_at timestamptz default now()
);

alter table share_links enable row level security;
alter table share_requests enable row level security;
alter table user_data enable row level security;

create policy "allow all" on share_links for all using (true) with check (true);
create policy "allow all" on share_requests for all using (true) with check (true);
create policy "allow all" on user_data for all using (true) with check (true);
